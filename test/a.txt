✅ 1️⃣ Final main.jsx
import React from 'react'
import ReactDOM from 'react-dom/client'
import { createBrowserRouter, RouterProvider } from 'react-router-dom'

import App from '@/App'                // Main layout with Navbar + Footer
import AuthLayout from '@/components/shared/AuthLayout'  // For Login/Signup (No Navbar/Footer)

import Home from '@/components/pages/Home'
import Login from '@/pages/Login'
import Signup from '@/pages/Signup'

const appRouter = createBrowserRouter([
  {
    path: "/",
    element: <App />,    // Layout with Navbar + Footer
    children: [
      { path: "home", element: <Home /> },    // /home
    ],
  },
  {
    path: "/login",
    element: <AuthLayout />,  // Layout without Navbar/Footer
    children: [
      { path: "", element: <Login /> },       // /login
    ],
  },
  {
    path: "/signup",
    element: <AuthLayout />,  // Layout without Navbar/Footer
    children: [
      { path: "", element: <Signup /> },      // /signup
    ],
  },
]);

ReactDOM.createRoot(document.getElementById('root')).render(
  <React.StrictMode>
    <RouterProvider router={appRouter} />
  </React.StrictMode>
);
